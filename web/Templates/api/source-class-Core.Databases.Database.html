 27 /**
 28  * This class is the base database class which is an extension of PDO
 29  *
 30  * @package Core\Database
 31  * @version $Revision$
 32  * @license http://creativecommons.org/licenses/by-sa/4.0/
 33  * @link http://coreframework.in
 34  * @author Shalom Sam <shalom.s@coreframework.in>
 35  */
 36 class Database extends PDO
 37 {
 38     /**
 39      * @var array Internal Caching
 40      */
 41     private $cache = [];
 42 
 43     /**
 44      * Creates an instance of the pdo class
 45      *
 46      * @param array $array
 47      * @throws \ErrorException
 48      * @throws \Exception
 49      */
 50     public function __construct(array $array = [])
 51     {
 52 
 53         if (empty($array)) {
 54             throw new \ErrorException("Connection parameters cannot be empty.");
 55         }
 56 
 57         if ($array['db'] == "") {
 58             $db = "";
 59         } elseif ($array['db'] != null) {
 60             $db = "dbname=" . $array['db'];
 61         }
 62 
 63         if (!empty($array['type'])) {
 64             $type = $array['type'];
 65         } else {
 66             $type = "mysql";
 67         }
 68 
 69         if (!empty($array['host'])) {
 70             $host = $array['host'];
 71         } else {
 72             throw new \Exception("Database Host not provided.");
 73         }
 74 
 75         if (!empty($array['username'])) {
 76             $user = $array['username'];
 77         } else {
 78             $user = null;
 79         }
 80 
 81         if (!empty($array['password'])) {
 82             $pass = $array['password'];
 83         } else {
 84             $pass = null;
 85         }
 86 
 87         if (!empty($array['port'])) {
 88             $port = "port=" . $array['port'];
 89         } else {
 90             $port = "";
 91         }
 92 
 93         $dsn = $type . ':' . $db . ';' . 'host=' . $host . ';' . $port;
 94 
 95         $dsnString = $dsn;
 96 
 97         if (isset($array['dsn'])) {
 98             $dsn = $array['type'] . ":";
 99             $dsnAttrArr = [];
100             foreach ($array['dsn'] as $key => $val) {
101                 $dsnAttrArr[] = $key . "=" . $val;
102             }
103 
104             $dsnAttr = implode(";", $dsnAttrArr);
105 
106             $dsnString = $dsn . $dsnAttr;
107         }
108 
109         return parent::__construct($dsnString, $user, $pass);
110     }
111 
112     /**
113      * Gets the prepared statement
114      *
115      * @param $query
116      * @return mixed
117      */
118     public function getPrepared($query)
119     {
120         $hash = md5($query);
121         if (!isset($this->cache[$hash])) {
122             $this->cache[$hash] = $this->prepare($query);
123         }
124         return $this->cache[$hash];
125     }
126 
127     /**
128      * Reset database cache
129      */
130     public function __destruct()
131     {
132         $this->cache = null;
133     }
134 
135 }
